stock StripAllWeapons(client)
{
	new iEnt;
	for (new i = 0; i <= 5; i++)
	{
		if (i != 2)
			while ((iEnt = GetPlayerWeaponSlot(client, i)) != -1)
			{
				RemovePlayerItem(client, iEnt);
				RemoveEdict(iEnt);
			}
	}
	//if (GetPlayerWeaponSlot(client, 2) == -1)    //Give player a knife/weapon
	//	GivePlayerItem(client, "weapon_knife");
}
